Cliffworld¶

Adapted from Example 6.6 on pg 132.

In [15]:
q_agent = Q_Agent(38, 4, epsilon=0.1, lr=0.05)
sarsa_agent = SARSA_Agent(38, 4, epsilon=0.1, lr=0.05)
env = CliffWorld()

epochs = 100
q_returns = np.zeros([epochs,500])
sarsa_returns = np.zeros([epochs,500])
for i in tqdm(range(epochs)):
    q_returns[i] = run_agent(env, q_agent, 500)
    sarsa_returns[i] = run_agent(env, sarsa_agent, 500)
